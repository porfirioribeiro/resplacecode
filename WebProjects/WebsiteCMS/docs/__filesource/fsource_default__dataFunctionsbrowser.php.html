  21.     Browser class
    22. 

  22.     
    23. 

  23.     Identifies the user's Operating system, Browser and version
    24. 

  24.     by parsing the HTTP_USER_AGENT string sent to the server
    25. 

  25.     
    26. 

  26.     Typical Usage:
    27. 

  27.     
    28. 

  28.         require_once($_SERVER['DOCUMENT_ROOT'].'/include/Browser.php');
    29. 

  29.         $br = new Browser;
    30. 

  30.         echo "$br->Platform, $br->Name version $br->Version";
    31. 

  31.     
    32. 

  32.     For operating systems, it will correctly identify:
    33. 

  33.         Microsoft Windows
    34. 

  34.         MacIntosh
    35. 

  35.         Linux
    36. 

  36.     Anything not determined to be one of the above is considered to by Unix
    37. 

  37.     because most Unix based Browsers seem to not report the operating system.
    38. 

  38.     The only known problem here is that, if a HTTP_USER_AGENT string does not
    39. 

  39.     contain the operating system, it will be identified as Unix. For unknown
    40. 

  40.     Browsers, this may not be correct.
    41. 

  41.     
    42. 

  42.     For Browsers, it should correctly identify all versions of:
    43. 

  43.         Amaya
    44. 

  44.         Galeon
    45. 

  45.         iCab
    46. 

  46.         Internet Explorer
    47. 

  47.             For AOL versions it will identify as Internet Explorer (AOL) and the version
    48. 

  48.             will be the AOL version instead of the IE version.
    49. 

  49.         Konqueror
    50. 

  50.         Lynx
    51. 

  51.         Mozilla
    52. 

  52.         Netscape Navigator/Communicator
    53. 

  53.         OmniWeb
    54. 

  54.         Opera
    55. 

  55.         Pocket Internet Explorer for handhelds
    56. 

  56.         Safari
    57. 

  57.         WebTV
    58. 

  58. *****************************************************************/

  61. class Browser{
    62. 

  62.  
    63. 

  63.     var $Name = "Unknown";
    64. 

  64.     var $Version = "Unknown";
    65. 

  65.     var $Platform = "Unknown";
    66. 

  66.     var $UserAgent = "Not reported";
    67. 

  67.     var $AOL = false;
    68. 

  68.  
    69. 

  69.     function Browser(){
    70. 

  70.         $agent = $_SERVER['HTTP_USER_AGENT'];
    71. 

  71.  
    72. 

  72.         // initialize properties
    73. 

  73.         $bd['platform'] = "Unknown";
    74. 

  74.         $bd['browser'] = "Unknown";
    75. 

  75.         $bd['version'] = "Unknown";
    76. 

  76.         $this->UserAgent = $agent;
    77. 

  77.  
    78. 

  78.         // find operating system
    79. 

  79.         if (eregi("win", $agent))
    80. 

  80.             $bd['platform'] = "Windows";
    81. 

  81.         elseif (eregi("mac", $agent))
    82. 

  82.             $bd['platform'] = "MacIntosh";
    83. 

  83.         elseif (eregi("linux", $agent))
    84. 

  84.             $bd['platform'] = "Linux";
    85. 

  85.         elseif (eregi("OS/2", $agent))
    86. 

  86.             $bd['platform'] = "OS/2";
    87. 

  87.         elseif (eregi("BeOS", $agent))
    88. 

  88.             $bd['platform'] = "BeOS";
    89. 

  89.  
    90. 

  90.         // test for Opera        
    91. 

  91.         if (eregi("opera",$agent)){
    92. 

  92.             $val = stristr($agent, "opera");
    93. 

  93.             if (eregi("/", $val)){
    94. 

  94.                 $val = explode("/",$val);
    95. 

  95.                 $bd['browser'] = $val[0];
    96. 

  96.                 $val = explode(" ",$val[1]);
    97. 

  97.                 $bd['version'] = $val[0];
    98. 

  98.             }else{
    99. 

  99.                 $val = explode(" ",stristr($val,"opera"));
    100. 

  100.                 $bd['browser'] = $val[0];
    101. 

  101.                 $bd['version'] = $val[1];
    102. 

  102.             }
    103. 

  103.  
    104. 

  104.         // test for WebTV
    105. 

  105.         }elseif(eregi("webtv",$agent)){
    106. 

  106.             $val = explode("/",stristr($agent,"webtv"));
    107. 

  107.             $bd['browser'] = $val[0];
    108. 

  108.             $bd['version'] = $val[1];
    109. 

  109.         
    110. 

  110.         // test for MS Internet Explorer version 1
    111. 

  111.         }elseif(eregi("microsoft internet explorer", $agent)){
    112. 

  112.             $bd['browser'] = "MSIE";
    113. 

  113.             $bd['version'] = "1.0";
    114. 

  114.             $var = stristr($agent, "/");
    115. 

  115.             if (ereg("308|425|426|474|0b1", $var)){
    116. 

  116.                 $bd['version'] = "1.5";
    117. 

  117.             }
    118. 

  118.  
    119. 

  119.         // test for NetPositive
    120. 

  120.         }elseif(eregi("NetPositive", $agent)){
    121. 

  121.             $val = explode("/",stristr($agent,"NetPositive"));
    122. 

  122.             $bd['platform'] = "BeOS";
    123. 

  123.             $bd['browser'] = $val[0];
    124. 

  124.             $bd['version'] = $val[1];
    125. 

  125.  
    126. 

  126.         // test for MS Internet Explorer
    127. 

  127.         }elseif(eregi("msie",$agent) && !eregi("opera",$agent)){
    128. 

  128.             $val = explode(" ",stristr($agent,"msie"));
    129. 

  129.             $bd['browser'] = $val[0];
    130. 

  130.             $bd['version'] = $val[1];
    131. 

  131.         
    132. 

  132.         // test for MS Pocket Internet Explorer
    133. 

  133.         }elseif(eregi("mspie",$agent) || eregi('pocket', $agent)){
    134. 

  134.             $val = explode(" ",stristr($agent,"mspie"));
    135. 

  135.             $bd['browser'] = "MSPIE";
    136. 

  136.             $bd['platform'] = "WindowsCE";
    137. 

  137.             if (eregi("mspie", $agent))
    138. 

  138.                 $bd['version'] = $val[1];
    139. 

  139.             else {
    140. 

  140.                 $val = explode("/",$agent);
    141. 

  141.                 $bd['version'] = $val[1];
    142. 

  142.             }
    143. 

  143.             
    144. 

  144.         // test for Galeon
    145. 

  145.         }elseif(eregi("galeon",$agent)){
    146. 

  146.             $val = explode(" ",stristr($agent,"galeon"));
    147. 

  147.             $val = explode("/",$val[0]);
    148. 

  148.             $bd['browser'] = $val[0];
    149. 

  149.             $bd['version'] = $val[1];
    150. 

  150.             
    151. 

  151.         // test for Konqueror
    152. 

  152.         }elseif(eregi("Konqueror",$agent)){
    153. 

  153.             $val = explode(" ",stristr($agent,"Konqueror"));
    154. 

  154.             $val = explode("/",$val[0]);
    155. 

  155.             $bd['browser'] = $val[0];
    156. 

  156.             $bd['version'] = $val[1];
    157. 

  157.             
    158. 

  158.         // test for iCab
    159. 

  159.         }elseif(eregi("icab",$agent)){
    160. 

  160.             $val = explode(" ",stristr($agent,"icab"));
    161. 

  161.             $bd['browser'] = $val[0];
    162. 

  162.             $bd['version'] = $val[1];
    163. 

  163.  
    164. 

  164.         // test for OmniWeb
    165. 

  165.         }elseif(eregi("omniweb",$agent)){
    166. 

  166.             $val = explode("/",stristr($agent,"omniweb"));
    167. 

  167.             $bd['browser'] = $val[0];
    168. 

  168.             $bd['version'] = $val[1];
    169. 

  169.  
    170. 

  170.         // test for Phoenix
    171. 

  171.         }elseif(eregi("Phoenix", $agent)){
    172. 

  172.             $bd['browser'] = "Phoenix";
    173. 

  173.             $val = explode("/", stristr($agent,"Phoenix/"));
    174. 

  174.             $bd['version'] = $val[1];
    175. 

  175.         
    176. 

  176.         // test for Firebird
    177. 

  177.         }elseif(eregi("firebird", $agent)){
    178. 

  178.             $bd['browser']="Firebird";
    179. 

  179.             $val = stristr($agent, "Firebird");
    180. 

  180.             $val = explode("/",$val);
    181. 

  181.             $bd['version'] = $val[1];
    182. 

  182.             
    183. 

  183.         // test for Firefox
    184. 

  184.         }elseif(eregi("Firefox", $agent)){
    185. 

  185.             $bd['browser']="Firefox";
    186. 

  186.             $val = stristr($agent, "Firefox");
    187. 

  187.             $val = explode("/",$val);
    188. 

  188.             $bd['version'] = $val[1];
    189. 

  189.             
    190. 

  190.       // test for Mozilla Alpha/Beta Versions
    191. 

  191.         }elseif(eregi("mozilla",$agent) && 
    192. 

  192.             eregi("rv:[0-9].[0-9][a-b]",$agent) && !eregi("netscape",$agent)){
    193. 

  193.             $bd['browser'] = "Mozilla";
    194. 

  194.             $val = explode(" ",stristr($agent,"rv:"));
    195. 

  195.             eregi("rv:[0-9].[0-9][a-b]",$agent,$val);
    196. 

  196.             $bd['version'] = str_replace("rv:","",$val[0]);
    197. 

  197.             
    198. 

  198.         // test for Mozilla Stable Versions
    199. 

  199.         }elseif(eregi("mozilla",$agent) &&
    200. 

  200.             eregi("rv:[0-9]\.[0-9]",$agent) && !eregi("netscape",$agent)){
    201. 

  201.             $bd['browser'] = "Mozilla";
    202. 

  202.             $val = explode(" ",stristr($agent,"rv:"));
    203. 

  203.             eregi("rv:[0-9]\.[0-9]\.[0-9]",$agent,$val);
    204. 

  204.             $bd['version'] = str_replace("rv:","",$val[0]);
    205. 

  205.         
    206. 

  206.         // test for Lynx & Amaya
    207. 

  207.         }elseif(eregi("libwww", $agent)){
    208. 

  208.             if (eregi("amaya", $agent)){
    209. 

  209.                 $val = explode("/",stristr($agent,"amaya"));
    210. 

  210.                 $bd['browser'] = "Amaya";
    211. 

  211.                 $val = explode(" ", $val[1]);
    212. 

  212.                 $bd['version'] = $val[0];
    213. 

  213.             } else {
    214. 

  214.                 $val = explode("/",$agent);
    215. 

  215.                 $bd['browser'] = "Lynx";
    216. 

  216.                 $bd['version'] = $val[1];
    217. 

  217.             }
    218. 

  218.         
    219. 

  219.         // test for Safari
    220. 

  220.         }elseif(eregi("safari", $agent)){
    221. 

  221.             $bd['browser'] = "Safari";
    222. 

  222.             $bd['version'] = "";
    223. 

  223.  
    224. 

  224.         // remaining two tests are for Netscape
    225. 

  225.         }elseif(eregi("netscape",$agent)){
    226. 

  226.             $val = explode(" ",stristr($agent,"netscape"));
    227. 

  227.             $val = explode("/",$val[0]);
    228. 

  228.             $bd['browser'] = $val[0];
    229. 

  229.             $bd['version'] = $val[1];
    230. 

  230.         }elseif(eregi("mozilla",$agent) && !eregi("rv:[0-9]\.[0-9]\.[0-9]",$agent)){
    231. 

  231.             $val = explode(" ",stristr($agent,"mozilla"));
    232. 

  232.             $val = explode("/",$val[0]);
    233. 

  233.             $bd['browser'] = "Netscape";
    234. 

  234.             $bd['version'] = $val[1];
    235. 

  235.         }
    236. 

  236.         
    237. 

  237.         // clean up extraneous garbage that may be in the name
    238. 

  238.         $bd['browser'] = ereg_replace("[^a-z,A-Z]", "", $bd['browser']);
    239. 

  239.         // clean up extraneous garbage that may be in the version        
    240. 

  240.         $bd['version'] = ereg_replace("[^0-9,.,a-z,A-Z]", "", $bd['version']);
    241. 

  241.         
    242. 

  242.         // check for AOL
    243. 

  243.         if (eregi("AOL", $agent)){
    244. 

  244.             $var = stristr($agent, "AOL");
    245. 

  245.             $var = explode(" ", $var);
    246. 

  246.             $bd['aol'] = ereg_replace("[^0-9,.,a-z,A-Z]", "", $var[1]);
    247. 

  247.         }
    248. 

  248.         
    249. 

  249.         // finally assign our properties
    250. 

  250.         $this->Name = $bd['browser'];
    251. 

  251.         $this->Version = $bd['version'];
    252. 

  252.         $this->Platform = $bd['platform'];
    253. 

  253.         $this->AOL = $bd['aol'];
    254. 

  254.     }
    255. 

  255. }
    256.
